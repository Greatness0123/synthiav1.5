/**
 * HumanoidMultiBodyManager — Multi-body Ragdoll with PD Motor Control
 *
 * Creates per-bone Rapier rigid bodies for the 15 major Mixamo bones,
 * connects them with impulse joints, and uses RapierJointMotorController
 * for PD motor control on 1-DOF joints (elbows, knees). Multi-DOF joints
 * (shoulders, hips, spine) use Spherical joints with angular damping.
 *
 * The AvatarSynchronizer copies physics RigidBody transforms to the
 * Three.js skeleton bones each frame. The existing capsule remains the
 * root body; limbs are attached via joints.
 *
 * Integration with HumanoidPhysicsBinder:
 *   - activate() → builds per-bone bodies and joints
 *   - setTargets() → sets PD motor target angles from currentTargets map
 *   - syncVisuals() → copies RigidBody transforms → skeleton bones
 *   - When active, HumanoidPhysicsBinder.updateMotorTargets() and
 *     syncVisuals() delegate to this manager instead of kinematic lerp.
 */

import * as THREE from 'three';
import RAPIER from '@dimforge/rapier3d-compat';
import { PhysicsEngine } from './PhysicsEngine';
import { RapierJointMotorController, JointMotorConfig } from './RapierJointMotorController';
import { AvatarSynchronizer } from './AvatarSynchronizer';
import { logger as Logger } from '../../utils/logger';
import { RAGDOLL_GROUP, ENVIRONMENT_GROUP, getCollisionMask } from '../../constants/physics';
import SYNTHIA_RIG_CONSTRAINTS from '../../constants/rigConstraints';
import { getAnatomicalLimitForBone } from '../../constants/anatomicalLimits';

// ── Shared quaternion/euler temporaries to avoid per-frame allocation ──
const _tempQuat = new THREE.Quaternion();
const _tempQuat2 = new THREE.Quaternion();
const _tempQuat3 = new THREE.Quaternion();
const _tempQuat4 = new THREE.Quaternion();
const _tempQuat5 = new THREE.Quaternion();
const _tempVec3 = new THREE.Vector3();
const _tempVec32 = new THREE.Vector3();
const _tempVec33 = new THREE.Vector3();

// ── Per-bone PD gain presets (from H1 teleop config + human2humanoid extraction) ──
// stiffness (Kp) = torque per radian of position error
// damping (Kd)   = torque per rad/s of velocity
interface PDGains {
  stiffness: number;
  damping: number;
}

const BONE_PD_GAINS: Record<string, PDGains> = {
  // Core spine — moderate stiffness for torso stability
  'mixamorigspine': { stiffness: 200, damping: 20 },
  'mixamorigneck': { stiffness: 100, damping: 10 },
  'mixamorighead':  { stiffness: 80,  damping: 8 },

  // Arms — lower stiffness allows natural arm swing
  'mixamorigleftarm':        { stiffness: 120, damping: 12 },
  'mixamorigrightarm':       { stiffness: 120, damping: 12 },
  'mixamorigleftforearm':    { stiffness: 80,  damping: 10 },
  'mixamorigrightforearm':   { stiffness: 80,  damping: 10 },
  'mixamoriglefthand':       { stiffness: 50,  damping: 5 },
  'mixamorigrighthand':      { stiffness: 50,  damping: 5 },

  // Legs — high stiffness for weight-bearing stability
  'mixamorigleftupleg':      { stiffness: 250, damping: 25 },
  'mixamorigrightupleg':     { stiffness: 250, damping: 25 },
  'mixamorigleftleg':        { stiffness: 250, damping: 25 },
  'mixamorigrightleg':       { stiffness: 250, damping: 25 },
  'mixamorigleftfoot':       { stiffness: 150, damping: 15 },
  'mixamorigrightfoot':      { stiffness: 150, damping: 15 },
};

// ── Joint type classification per bone ──
type JointType = 'revolute' | 'spherical';
const BONE_JOINT_TYPE: Record<string, JointType> = {
  'mixamorigspine':           'spherical',
  'mixamorigneck':            'spherical',
  'mixamorighead':            'spherical',
  'mixamorigleftarm':         'spherical',
  'mixamorigrightarm':        'spherical',
  'mixamorigleftforearm':     'revolute',
  'mixamorigrightforearm':    'revolute',
  'mixamoriglefthand':        'spherical',
  'mixamorigrighthand':       'spherical',
  'mixamorigleftupleg':       'spherical',
  'mixamorigrightupleg':      'spherical',
  'mixamorigleftleg':         'revolute',
  'mixamorigrightleg':        'revolute',
  'mixamorigleftfoot':        'spherical',
  'mixamorigrightfoot':       'spherical',
};

// Bones that attach to the capsule root instead of their visual parent (hips bone).
// In the Three.js scene graph, these bones' parent is mixamorig:Hips,
// but the physics system uses the capsule as the root body.
const CAPSULE_ATTACH_BONES = new Set([
  'mixamorigspine', 'mixamorigleftupleg', 'mixamorigrightupleg',
]);

/**
 * Derive parent canonical name from bone.parent.
 * Returns null when the bone should attach to the capsule root.
 */
function getPhysicsParentName(bone: THREE.Bone, trackedBones: Set<string>): string | null {
  const canonical = bone.name.toLowerCase().replace(/:/g, '');
  if (CAPSULE_ATTACH_BONES.has(canonical)) return null;
  let parent: THREE.Object3D | null = bone.parent;
  while (parent) {
    if (parent instanceof THREE.Bone) {
      const parentCanonical = parent.name.toLowerCase().replace(/:/g, '');
      if (trackedBones.has(parentCanonical)) return parentCanonical;
    }
    parent = parent.parent;
  }
  return null; // fallback → capsule
}

/**
 * Topologically sort bones so parents are processed before children.
 */
function topologicalSortBones(
  bones: THREE.Bone[],
  trackedBones: Set<string>
): THREE.Bone[] {
  const sorted: THREE.Bone[] = [];
  const visited = new Set<string>();

  function visit(bone: THREE.Bone) {
    const canonical = bone.name.toLowerCase().replace(/:/g, '');
    if (visited.has(canonical)) return;
    visited.add(canonical);
    const parentName = getPhysicsParentName(bone, trackedBones);
    if (parentName) {
      const parentBone = bones.find(
        b => b.name.toLowerCase().replace(/:/g, '') === parentName
      );
      if (parentBone) visit(parentBone);
    }
    sorted.push(bone);
  }

  for (const bone of bones) visit(bone);
  return sorted;
}

// Revolute axis for 1-DOF joints (elbow/knee — local X axis in Mixamo bone space)
const REVOLUTE_AXIS = { x: 1, y: 0, z: 0 };

interface PerBoneData {
  canonicalName: string;
  rigidBody: RAPIER.RigidBody;
  collider: RAPIER.Collider;
  joint: RAPIER.ImpulseJoint | null;
}

export class HumanoidMultiBodyManager {
  private physicsEngine: PhysicsEngine;
  private world: RAPIER.World;
  private scene: THREE.Scene;

  private motorController: RapierJointMotorController;
  private avatarSynchronizer: AvatarSynchronizer;

  private bodies: Map<string, PerBoneData> = new Map();
  private rigidBodiesMap: Map<string, RAPIER.RigidBody> = new Map();

  // Reference to the capsule root body (owned by HumanoidPhysicsBinder)
  private capsuleBody: RAPIER.RigidBody | null = null;

  // Stored references for runtime parent lookups
  private boneInfoMap: Map<string, { bone: THREE.Bone; worldPosition: THREE.Vector3 }> | null = null;
  private trackedBones: Set<string> = new Set();

  // Bind pose relative quaternions: child bone's local rotation at bind pose
  // (i.e., parentBindPose.invert() * childBindPose). Used to convert AI deltas
  // (which are relative to bind pose) into absolute target rotations for PD control.
  private bindPoseRelativeQuats: Map<string, THREE.Quaternion> = new Map();

  // Model parameters needed for positioning
  private capsuleCenterY: number = 0;
  private modelRoot: THREE.Group | null = null;

  // State
  public isActive: boolean = false;

  // PD gain adjustment
  private globalStiffnessScale: number = 1.0;
  private globalDampingScale: number = 1.0;

  constructor(physicsEngine: PhysicsEngine, scene: THREE.Scene) {
    this.physicsEngine = physicsEngine;
    this.world = physicsEngine.getWorld();
    this.scene = scene;
    this.motorController = new RapierJointMotorController();
    this.avatarSynchronizer = new AvatarSynchronizer(0.04);
  }

  /**
   * Build all per-bone rigid bodies and impulse joints.
   * Called once after HumanoidPhysicsBinder has reached STEP D.
   */
  public async activate(
    boneInfoMap: Map<string, { bone: THREE.Bone; worldPosition: THREE.Vector3 }>,
    _skeleton: THREE.Skeleton,
    capsuleBody: RAPIER.RigidBody,
    capsuleCenterY: number,
    modelRoot: THREE.Group
  ): Promise<boolean> {
    void this.scene; // silence unused warning — kept for future debug visualization
    if (this.isActive) return true;

    this.physicsEngine.setMutating(true);

    try {
      this.capsuleBody = capsuleBody;
      this.capsuleCenterY = capsuleCenterY;
      this.modelRoot = modelRoot;

      // Clean any previous state
      this.deactivate();

      // Build tracked bones set for parent lookups
      const trackedBones = new Set<string>();
      for (const canonical of boneInfoMap.keys()) {
        if (BONE_JOINT_TYPE[canonical]) {
          trackedBones.add(canonical);
        }
      }
      this.boneInfoMap = boneInfoMap;
      this.trackedBones = trackedBones;

      // Topologically sort bones so parents are processed before children
      const allBones = Array.from(boneInfoMap.values()).map(info => info.bone);
      const sortedBones = topologicalSortBones(allBones, trackedBones);

      // Build each bone in hierarchy order (parent first)
      for (const bone of sortedBones) {
        const boneName = bone.name.toLowerCase().replace(/:/g, '');
        const parentName = getPhysicsParentName(bone, trackedBones);
        const boneInfo = boneInfoMap.get(boneName);
        if (!boneInfo) {
          Logger.warn(`HMB: Bone ${boneName} not found, skipping`);
          continue;
        }

        // Bone world position in physics space
        // The bone's worldPosition is in visual space; we need to align it
        // with the capsule root which may have moved.
        const boneWorldPos = boneInfo.worldPosition.clone();

        // Create rigid body at the bone's world position
        const rbDesc = RAPIER.RigidBodyDesc.dynamic()
          .setTranslation(boneWorldPos.x, boneWorldPos.y, boneWorldPos.z)
          .setGravityScale(0)
          .setLinearDamping(10.0)
          .setAngularDamping(20.0);
        const body = this.world.createRigidBody(rbDesc);

        // Collider: small capsule sized to bone segment
        const boneLength = this.estimateBoneLength(boneName, boneInfo, boneInfoMap);
        const colRadius = 0.04;
        const colHalfHeight = Math.max(0.02, boneLength / 2 - colRadius);
        const colDesc = RAPIER.ColliderDesc.capsule(colHalfHeight, colRadius)
          .setCollisionGroups(getCollisionMask(RAGDOLL_GROUP, ENVIRONMENT_GROUP))
          .setActiveEvents(RAPIER.ActiveEvents.CONTACT_FORCE_EVENTS);

        const collider = this.world.createCollider(colDesc, body);
        this.physicsEngine.registerVelocityClampBody(body);

        this.rigidBodiesMap.set(boneName, body);

        // Determine parent body: either another bone's body or the capsule
        let parentBody: RAPIER.RigidBody | null = null;
        if (parentName) {
          parentBody = this.rigidBodiesMap.get(parentName) ?? null;
        }
        if (!parentBody) {
          parentBody = capsuleBody; // parent = capsule for root-level bones
        }
        if (!parentBody || !parentBody.isValid()) {
          Logger.warn(`HMB: Invalid parent for ${boneName}, storing as root-level body`);
          this.bodies.set(boneName, { canonicalName: boneName, rigidBody: body, collider, joint: null });
          continue;
        }

        // Create joint between parent and this bone
        const jointType = BONE_JOINT_TYPE[boneName] ?? 'spherical';
        let joint: RAPIER.ImpulseJoint;

        // Joint anchor: at this bone's position in each body's local frame
        const childPos = body.translation();
        const parentPos = parentBody.translation();

        // Anchor1 = position of child in parent body's local frame
        const anchor1 = {
          x: childPos.x - parentPos.x,
          y: childPos.y - parentPos.y,
          z: childPos.z - parentPos.z,
        };
        // Anchor2 = local origin of child body (0, 0, 0 relative to itself)
        const anchor2 = { x: 0, y: 0, z: 0 };

        if (jointType === 'revolute') {
          const jointData = RAPIER.JointData.revolute(anchor1, anchor2, REVOLUTE_AXIS);

          // Set joint limits from SYNTHIA_RIG_CONSTRAINTS
          const constraint = SYNTHIA_RIG_CONSTRAINTS[boneName];
          if (constraint && constraint.dof === 1) {
            const min = constraint.x[0];
            const max = constraint.x[1];
            if (isFinite(min) && isFinite(max)) {
              jointData.limitsEnabled = true;
              jointData.limits = [min, max];
            }
          } else {
            // Fallback: use anatomical limits
            const limits = getAnatomicalLimitForBone(boneName);
            if (limits) {
              jointData.limitsEnabled = true;
              jointData.limits = [limits.min, limits.max];
            }
          }

          joint = this.world.createImpulseJoint(jointData, parentBody, body, true);
        } else {
          // Spherical joint for multi-DOF bones (shoulder, hip, spine, neck, wrist, ankle)
          const jointData = RAPIER.JointData.spherical(anchor1, anchor2);

          // Set swing/twist limits
          const constraint = SYNTHIA_RIG_CONSTRAINTS[boneName];
          if (constraint && constraint.dof >= 2) {
            // Use the most restrictive axis for spherical joint limit
            const primaryLimit = Math.min(
              Math.abs(constraint.x[1] - constraint.x[0]) / 2,
              Math.abs(constraint.y[1] - constraint.y[0]) / 2,
              Math.abs(constraint.z[1] - constraint.z[0]) / 2
            );
            if (isFinite(primaryLimit) && primaryLimit > 0.01) {
              jointData.limitsEnabled = true;
              jointData.limits = [-primaryLimit, primaryLimit];
            }
          }

          joint = this.world.createImpulseJoint(jointData, parentBody, body, true);
        }

        this.bodies.set(boneName, { canonicalName: boneName, rigidBody: body, collider, joint });
      }

      // ── Compute bind pose relative quaternions for PD targeting ───────
      // AI targets are deltas from bind pose. The PD controller computes:
      //   currentRelativeQuat = parentWorld⁻¹ * childWorld = childLocalQuat
      // So the target must be in childLocal space: childBindLocal * deltaQuat.
      this.bindPoseRelativeQuats.clear();
      for (const [boneName, boneInfo] of boneInfoMap) {
        if (!BONE_JOINT_TYPE[boneName]) continue;
        const bone = boneInfo.bone;
        // Store the bone's local quaternion at bind pose
        this.bindPoseRelativeQuats.set(boneName, bone.quaternion.clone());
      }

      // ── Register joints with motor controller ──
      const motorConfigs: JointMotorConfig[] = [];

      for (const [boneName, bodyData] of this.bodies) {
        if (!bodyData.joint) continue;

        const gains = BONE_PD_GAINS[boneName] ?? { stiffness: 100, damping: 10 };
        const jointType = BONE_JOINT_TYPE[boneName] ?? 'spherical';

        const constraint = SYNTHIA_RIG_CONSTRAINTS[boneName];
        let limits: [number, number] | undefined;
        if (constraint && constraint.dof === 1) {
          limits = [constraint.x[0], constraint.x[1]];
        } else {
          const anatLim = getAnatomicalLimitForBone(boneName);
          if (anatLim) limits = [anatLim.min, anatLim.max];
        }

        motorConfigs.push({
          name: boneName,
          joint: bodyData.joint,
          jointType: jointType === 'revolute' ? RAPIER.JointType.Revolute : RAPIER.JointType.Spherical,
          stiffness: gains.stiffness,
          damping: gains.damping,
          axisIndex: 0,
          dof: jointType === 'revolute' ? 1 : 3,
          limits,
          targetAngle: 0,
        });
      }

      this.motorController.registerJoints(motorConfigs);

      // ── Register bones with AvatarSynchronizer ──
      this.avatarSynchronizer.clear();
      for (const [boneName] of this.bodies) {
        const boneInfo = boneInfoMap.get(boneName);
        if (!boneInfo) continue;

        this.avatarSynchronizer.registerBone(
          boneName,
          boneInfo.bone.name,
          {
            canonicalName: boneName,
            syncRotation: true,
            syncTranslation: false,
            rootOffsetY: 0,
          }
        );
      }

      // ── Set all revolute joints to bind-pose zero (upright stance) ──
      // Call after registerJoints so the motor controller can apply targets.
      // Spherical joints are handled by manual PD torque in setTargets(), not
      // by Rapier's built-in motor — skip them to avoid spurious warnings.
      this.motorController.getJointNames().forEach((name) => {
        if (BONE_JOINT_TYPE[name] === 'revolute') {
          this.motorController.setTargetAngle(name, 0);
        }
      });

      this.isActive = true;

      Logger.info(`HumanoidMultiBodyManager: Activated with ${this.bodies.size} bone bodies, ${motorConfigs.length} motor joints`);
      return true;

    } catch (error) {
      Logger.error('HumanoidMultiBodyManager: Activation failed', error);
      this.deactivate();
      return false;
    } finally {
      this.physicsEngine.setMutating(false);
      this.physicsEngine.setReady(true);
    }
  }

  /**
   * Set motor targets from the HumanoidPhysicsBinder's currentTargets map.
   * Called every frame when multi-body is active.
   * Translates [pitch, yaw, roll] Euler targets into PD motor positions.
   *
   * For revolute joints (elbows, knees): uses Rapier's built-in PD motor.
   * For spherical joints (shoulders, hips, spine, neck): applies manual PD
   * torque since Rapier v0.19 SphericalImpulseJoint lacks motor support.
   */
  public setTargets(currentTargets: Map<string, any>): void {
    if (!this.isActive) return;

    currentTargets.forEach((parsedTarget, canonical) => {
      const bodyData = this.bodies.get(canonical);
      if (!bodyData) return;

      const jointType = BONE_JOINT_TYPE[canonical];
      const gains = BONE_PD_GAINS[canonical] ?? { stiffness: 100, damping: 10 };

      if (jointType === 'revolute' && bodyData.joint) {
        // ── Revolute Joint PD Control (Rapier built-in motor) ──
        let targetAngle = 0;
        if (parsedTarget.isScalar && typeof parsedTarget.scalar === 'number') {
          targetAngle = parsedTarget.scalar;
        } else if (parsedTarget.x !== undefined && typeof parsedTarget.x === 'number') {
          targetAngle = parsedTarget.x;
        }
        // Clamp to limits
        const constraint = SYNTHIA_RIG_CONSTRAINTS[canonical];
        if (constraint && constraint.dof === 1) {
          targetAngle = Math.max(constraint.x[0], Math.min(constraint.x[1], targetAngle));
        }
        this.motorController.setTargetAngle(canonical, targetAngle);

      } else if (jointType === 'spherical' && bodyData.rigidBody.isValid()) {
        // ── Spherical Joint Quaternion-Based PD Torque ────────────────────
        // Rapier v0.19 SphericalImpulseJoint has no configureMotorPosition().
        // We compute PD torque manually and apply it to the child body.
        //
        // Quaternion-based PD avoids Euler cross-coupling that causes limit
        // cycle oscillation at large rotations (e.g. 75° arms-down).
        //
        // PD equation (axis-angle): τ = Kp * (axis * angle) - Kd * ω_local
        // ──────────────────────────────────────────────────────────────────

        // Build target quaternion from parsed target
        // AI targets are deltas from bind pose. Multiply by bindPoseRelative
        // to get the absolute target rotation in parent-local space.
        const bindPoseRelative = this.bindPoseRelativeQuats.get(canonical);
        const deltaQuat = _tempQuat5.identity();
        if (parsedTarget.isScalar && typeof parsedTarget.scalar === 'number') {
          // X-axis (Pitch) for all bones including arms.
          deltaQuat.setFromAxisAngle(_tempVec3.set(1, 0, 0), parsedTarget.scalar);
        } else if (parsedTarget.x !== undefined) {
          // XYZ Euler order for all bones (X-axis is Pitch for arm adduction).
          const euler = new THREE.Euler(parsedTarget.x || 0, parsedTarget.y || 0, parsedTarget.z || 0, 'XYZ');
          deltaQuat.setFromEuler(euler);
        }

        // targetQuat = bindPoseRelative * deltaFromBindPose
        const targetQuat = bindPoseRelative
          ? _tempQuat2.copy(bindPoseRelative).multiply(deltaQuat)
          : _tempQuat2.copy(deltaQuat);

        // Get parent body from bone.parent hierarchy
        const boneObj = this.boneInfoMap?.get(canonical)?.bone;
        const parentName = boneObj ? getPhysicsParentName(boneObj, this.trackedBones) : null;
        const parentBody = parentName ? this.rigidBodiesMap.get(parentName) ?? this.capsuleBody : this.capsuleBody;

        if (!parentBody || !parentBody.isValid()) return;

        // Current relative rotation (parent frame → child frame)
        const childRot = bodyData.rigidBody.rotation();
        const parentRot = parentBody.rotation();
        const childQuat = _tempQuat.set(childRot.x, childRot.y, childRot.z, childRot.w);
        const parentQuat = _tempQuat3.set(parentRot.x, parentRot.y, parentRot.z, parentRot.w);

        _tempQuat.copy(parentQuat).invert().multiply(childQuat);

        // Error quaternion: how far current is from target
        // errorQuat = targetQuat * currentQuat.inverse()
        const errorQuat = _tempQuat4.copy(targetQuat).multiply(_tempQuat.invert());

        // Convert error to axis-angle (single rotation, no Euler cross-coupling)
        const errorAxis = new THREE.Vector3();
        let errorAngle = 0;
        errorQuat.normalize();
        errorAxis.set(errorQuat.x, errorQuat.y, errorQuat.z);
        const sinHalf = errorAxis.length();
        if (sinHalf > 1e-6) {
          errorAngle = 2 * Math.atan2(sinHalf, errorQuat.w);
          errorAxis.divideScalar(sinHalf);
        } else {
          errorAngle = 0;
          errorAxis.set(0, 1, 0);
        }

        // Torque in local frame: τ = Kp * axis * angle - Kd * angularVelocity_local
        const angVel = bodyData.rigidBody.angvel();
        const localAngVel = _tempVec3.set(angVel.x, angVel.y, angVel.z)
          .applyQuaternion(parentQuat.clone().invert());

        const stiffness = gains.stiffness * this.globalStiffnessScale;
        const damping = gains.damping * this.globalDampingScale;

        const localTorque = _tempVec3.set(
          stiffness * errorAxis.x * errorAngle - damping * localAngVel.x,
          stiffness * errorAxis.y * errorAngle - damping * localAngVel.y,
          stiffness * errorAxis.z * errorAngle - damping * localAngVel.z
        );

        // Clamp magnitude to prevent explosive forces
        const torqueMag = localTorque.length();
        const MAX_TORQUE = 50.0;
        if (torqueMag > MAX_TORQUE) {
          localTorque.multiplyScalar(MAX_TORQUE / torqueMag);
        }

        // Transform to world frame and apply
        bodyData.rigidBody.addTorque(
          localTorque.applyQuaternion(parentQuat),
          true
        );
      }
    });

    // ── Capsule Upright Spring (Balance Controller) ─────────────────────────
    // Apply a restoring torque to the capsule body to keep it vertical.
    // Prevents the capsule from toppling when bone torques create net moment.
    // PD equation: τ = Kp * (tilt_axis * tilt_angle) - Kd * ω_capsule
    // ────────────────────────────────────────────────────────────────────────
    if (this.capsuleBody && this.capsuleBody.isValid()) {
      const capsuleQuat = this.capsuleBody.rotation();
      _tempQuat.set(capsuleQuat.x, capsuleQuat.y, capsuleQuat.z, capsuleQuat.w);

      // Extract tilt: how far the capsule's local Y-axis deviates from world up (0,1,0)
      const capsuleUp = _tempVec3.set(0, 1, 0).applyQuaternion(_tempQuat);
      const tiltAngle = Math.acos(Math.min(1, Math.max(-1, capsuleUp.y)));
      const tiltAxis = _tempVec32.set(-capsuleUp.z, 0, capsuleUp.x).normalize();

      // Balance PD gains — tuned for ~80kg humanoid capsule
      const BALANCE_KP = 80.0;
      const BALANCE_KD = 30.0;

      const angVel = this.capsuleBody.angvel();
      const balanceTorque = _tempVec33.set(
        BALANCE_KP * tiltAxis.x * tiltAngle - BALANCE_KD * angVel.x,
        BALANCE_KP * tiltAxis.y * tiltAngle - BALANCE_KD * angVel.y,
        BALANCE_KP * tiltAxis.z * tiltAngle - BALANCE_KD * angVel.z
      );

      // Clamp to prevent explosive forces
      const torqueMag = balanceTorque.length();
      const MAX_BALANCE_TORQUE = 100.0;
      if (torqueMag > MAX_BALANCE_TORQUE) {
        balanceTorque.multiplyScalar(MAX_BALANCE_TORQUE / torqueMag);
      }

      this.capsuleBody.addTorque(balanceTorque, true);
    }
  }

  /**
   * Sync physics RigidBody transforms to visual skeleton bones.
   * Called every frame from HumanoidPhysicsBinder.syncVisuals().
   */
  public syncVisuals(
    boneInfoMap: Map<string, { bone: THREE.Bone; worldPosition: THREE.Vector3 }>,
    _skeleton: THREE.Skeleton,
    disableSync?: Set<string>
  ): void {
    if (!this.isActive) return;

    // Build a map of canonical name → THREE.Bone position data for AvatarSynchronizer
    const bonesSyncMap = new Map<string, { bone: THREE.Bone; worldPosition: THREE.Vector3 }>();

    for (const [canonical] of this.bodies) {
      const boneInfo = boneInfoMap.get(canonical);
      if (!boneInfo) continue;

      const worldPos = new THREE.Vector3();
      boneInfo.bone.getWorldPosition(worldPos);

      bonesSyncMap.set(canonical, {
        bone: boneInfo.bone,
        worldPosition: worldPos,
      });
    }

    // Sync RigidBody rotation → bone quaternion
    this.avatarSynchronizer.synchronize(
      bonesSyncMap,
      this.rigidBodiesMap,
      disableSync
    );

    // Also sync root capsule → modelRoot
    if (this.capsuleBody && this.capsuleBody.isValid() && this.modelRoot) {
      this.avatarSynchronizer.syncRoot(
        this.modelRoot,
        this.capsuleBody,
        this.capsuleCenterY
      );
    }
  }

  /**
   * Set global PD gain multipliers.
   * 1.0 = nominal, >1.0 = stiffer, <1.0 = looser.
   */
  public setGainScale(stiffnessScale: number, dampingScale: number): void {
    this.globalStiffnessScale = Math.max(0.01, stiffnessScale);
    this.globalDampingScale = Math.max(0.01, dampingScale);

    // Re-apply gains to all joints
    this.motorController.getJointNames().forEach((name) => {
      const base = BONE_PD_GAINS[name];
      if (base) {
        this.motorController.setGains(
          name,
          base.stiffness * this.globalStiffnessScale,
          base.damping * this.globalDampingScale
        );
      }
    });
  }

  /**
   * Toggle limp mode (ragdoll: zero effective stiffness).
   */
  public setLimpMode(limp: boolean): void {
    this.motorController.setLimpMode(limp);
  }

  /**
   * Get the RapierJointMotorController for direct access.
   */
  public getMotorController(): RapierJointMotorController {
    return this.motorController;
  }

  /**
   * Get the AvatarSynchronizer for direct access.
   */
  public getAvatarSynchronizer(): AvatarSynchronizer {
    return this.avatarSynchronizer;
  }

  /**
   * Snap all RigidBody transforms to match the current skeleton bone quaternions.
   * Called after resetToBindPose() so that syncVisuals() copies the correct
   * (reset) values back to bones instead of stale RigidBody rotations.
   *
   * Uses getWorldQuaternion() to compute the world-space rotation from the full
   * bone hierarchy chain, then sets it on the RigidBody (which stores world-space
   * transforms). Also zeros velocity to prevent residual momentum from causing
   * drift on the first frame after reset.
   */
  public syncRigidBodiesFromBones(
    boneInfoMap: Map<string, { bone: THREE.Bone; worldPosition: THREE.Vector3 }>
  ): void {
    if (!this.isActive) return;

    // Ensure scene graph matrices are up to date so world quaternions are correct
    if (this.modelRoot) {
      this.modelRoot.updateMatrixWorld(true);
    }

    for (const [canonical, bodyData] of this.bodies) {
      const boneInfo = boneInfoMap.get(canonical);
      if (!boneInfo || !bodyData.rigidBody.isValid()) continue;

      // Compute WORLD quaternion from the full bone hierarchy chain
      const worldQuat = new THREE.Quaternion();
      boneInfo.bone.getWorldQuaternion(worldQuat);

      bodyData.rigidBody.setRotation(
        { x: worldQuat.x, y: worldQuat.y, z: worldQuat.z, w: worldQuat.w },
        true
      );

      // Also teleport position to match bone world position
      const worldPos = new THREE.Vector3();
      boneInfo.bone.getWorldPosition(worldPos);
      bodyData.rigidBody.setTranslation(
        { x: worldPos.x, y: worldPos.y, z: worldPos.z },
        true
      );

      bodyData.rigidBody.setLinvel({ x: 0, y: 0, z: 0 }, true);
      bodyData.rigidBody.setAngvel({ x: 0, y: 0, z: 0 }, true);
    }
  }

  /**
   * Get the number of active bone bodies.
   */
  public getBoneCount(): number {
    return this.bodies.size;
  }

  /**
   * Check if the multi-body system is active.
   */
  public getIsActive(): boolean {
    return this.isActive;
  }

  /**
   * Get the rigid bodies map for ObservationBuilder registration.
   */
  public getRigidBodiesMap(): Map<string, RAPIER.RigidBody> {
    return this.rigidBodiesMap;
  }

  /**
   * Get the capsule root body reference.
   */
  public getCapsuleBody(): RAPIER.RigidBody | null {
    return this.capsuleBody;
  }

  /**
   * Get the collider handle for a bone by canonical name.
   * Returns null if the bone has no collider (e.g. finger bones, or multi-body inactive).
   */
  public getBoneColliderHandle(canonicalName: string): number | null {
    const bodyData = this.bodies.get(canonicalName);
    if (!bodyData || !bodyData.collider) return null;
    return bodyData.collider.handle;
  }

  /**
   * Remove all per-bone rigid bodies, joints, and colliders.
   */
  public deactivate(): void {
    if (!this.isActive && this.bodies.size === 0) return;

    this.physicsEngine.setMutating(true);

    try {
      this.bodies.forEach((data) => {
        try {
          if (data.rigidBody.isValid()) {
            this.physicsEngine.unregisterVelocityClampBody(data.rigidBody);
            this.world.removeRigidBody(data.rigidBody);
          }
        } catch (e) {
          // Silently ignore removal errors during cleanup
        }
      });

      this.bodies.clear();
      this.rigidBodiesMap.clear();
      this.boneInfoMap = null;
      this.trackedBones.clear();
      this.bindPoseRelativeQuats.clear();
      this.motorController.cleanup();
      this.avatarSynchronizer.clear();
      this.isActive = false;
      Logger.info('HumanoidMultiBodyManager: Deactivated');
    } finally {
      this.physicsEngine.setMutating(false);
      this.physicsEngine.setReady(true);
    }
  }

  /**
   * Estimate bone segment length from bone name and position.
   * Uses real bone positions when available, falls back to heuristic.
   */
  private estimateBoneLength(
    boneName: string,
    boneInfo: { bone: THREE.Bone; worldPosition: THREE.Vector3 },
    allBones: Map<string, { bone: THREE.Bone; worldPosition: THREE.Vector3 }>
  ): number {
    // Use bone.children to find the first tracked child bone for distance measurement
    const firstChild = boneInfo.bone.children.find((child): child is THREE.Bone => {
      if (!(child instanceof THREE.Bone)) return false;
      const childCanonical = child.name.toLowerCase().replace(/:/g, '');
      return allBones.has(childCanonical);
    });

    if (firstChild) {
      const childCanonical = firstChild.name.toLowerCase().replace(/:/g, '');
      const childInfo = allBones.get(childCanonical);
      if (childInfo) {
        const dx = childInfo.worldPosition.x - boneInfo.worldPosition.x;
        const dy = childInfo.worldPosition.y - boneInfo.worldPosition.y;
        const dz = childInfo.worldPosition.z - boneInfo.worldPosition.z;
        return Math.sqrt(dx * dx + dy * dy + dz * dz);
      }
    }

    // Parent-based fallback for bones where we know approximate lengths
    const heuristicLengths: Record<string, number> = {
      'mixamorigspine': 0.25,
      'mixamorigneck': 0.10,
      'mixamorighead': 0.12,
      'mixamorigleftarm': 0.30,
      'mixamorigrightarm': 0.30,
      'mixamorigleftforearm': 0.27,
      'mixamorigrightforearm': 0.27,
      'mixamoriglefthand': 0.10,
      'mixamorigrighthand': 0.10,
      'mixamorigleftupleg': 0.42,
      'mixamorigrightupleg': 0.42,
      'mixamorigleftleg': 0.40,
      'mixamorigrightleg': 0.40,
      'mixamorigleftfoot': 0.12,
      'mixamorigrightfoot': 0.12,
    };

    return heuristicLengths[boneName] ?? 0.15;
  }
}
